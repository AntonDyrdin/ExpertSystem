        if(dataset[i + window_size,bid_column_index]) > (dataset[i + window_size-1,ask_column_index]):

        else:
        #   если на следующем шаге графика  аск (цена предложения) упадёт ниже, чем бид сейчас (цена спроса), то Y=[0;0;1] - выход с рынка
            if(dataset[i + window_size,ask_column_index]) < (dataset[i + window_size-1,bid_column_index]):
                Dataset_Y[i,0] = 0
                Dataset_Y[i,1] = 1

            else:
